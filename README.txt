To compile & run make sure PATH_TO_FX is set to your javafx lib folder location

compile command:
javac --module-path %PATH_TO_FX% --add-modules javafx.controls,javafx.fxml,javafx.graphics,javafx.controls "filename.java"


run command:

java --module-path %PATH_TO_FX% --add-modules javafx.controls,javafx.fxml,javafx.graphics,javafx.controls "filename"
